			output.writeUint32(0);

			output.writeUint32(size);
			output.cursor += size;

    MIDIMessage (track, time, message)

  • ¶

    MIDI message abstraction class.

    track is the track that contains this event.

    time is the time (relative to the previous event) this message is to be interpreted.

    message is an object describing the MIDI message, of the form:

    {
    type,
    subtype,
    channel,
    data
}

    type is the message type, one of TinySMF.META, TinySMF.SYSEX, TinySMF.SYSEX_CONT, or TinySMF.CHANNEL.

    subtype describes the precise type of event, e.g. TinySMF.Meta.TrackName or TinySMF.Channel.NoteOn

    channel is only relevant for TinySMF.CHANNEL events, it contains the channel this message applies to. If omitted, 0 is assumed.

    data is an array, its elements vary depending on the needs of the message subtype. For example, TinySMF.Channel.NoteOn data contains 2 bytes representing the pitch of the note, and its velocity.

    I’d recommend reading up on the different types of messages and their corresponding data bytes.

    	function MIDIMessage (track, time, message) {
		this.track = track;

		this.time = time || 0;

		if (typeof message !== "undefined") {
			this.type = message.type;
			this.subtype = message.subtype;

			if (message.type == TinySMF.CHANNEL)
				this.channel = message.channel || 0;

			this.data = message.data;
		} else {
			this.type = TinySMF.SYSEX;
			this.subtype = 0;
			this.data = [0x7f];
		}
	}

    MIDIMessage.write (output)

  • ¶

    Writes this MIDIMessage to MetaView instance output.

    		write: function (output) {
			VLQ.write(output, this.time);

			if (this.type == TinySMF.META) {
				output.writeUint8(0xFF);
				output.writeUint8(this.subtype);
				VLQ.write(output, this.data.length);
				if (this.data.length)
					output.writeUint8(this.data);
			} else if (this.type == TinySMF.SYSEX ||
								 this.type == TinySMF.SYSEX_CONT) {
				output.writeUint8(this.status);
				VLQ.write(output, this.data.length);
				if (this.data.length)
					output.writeUint8(this.data);
			} else if (this.type == TinySMF.CHANNEL){
				output.writeUint8((this.subtype << 4) | this.channel);
				if (this.subtype >= 0x8 && this.subtype <= 0xB) {

				} else if (this.subtype == 0xE) {

    		read: function (input) {
			var chunkID, chunk;

			this.header = new MIDIHeader();
			this.header.read(input);

			while (this.tracks.length != this.header.tracks) {
				chunkID = toASCII(input.peekUint8(4));

				if (chunkID == TinySMF.TrackChunk) {
					chunk = new TinySMF.MIDITrack();
					chunk.read(input);
					this.tracks.push(chunk);
